var moveZeroes = function (nums) {
  let start = nums.length - 1;

  while (start >= 0) {
    if (nums[start] === 0) {
      for (let i = start; i < nums.length - 1; i++) {
        const tmp = nums[i];
        nums[i] = nums[i + 1];
        nums[i + 1] = tmp;
      }
    }
    start--;
  }
};

function moveZeroes(nums) {
  let i = 0;
  for (let j = 0; j < nums.length; j++) {
    if (nums[j] !== 0) {
      nums[i] = nums[j];
      i++;
    }
  }
  for (let k = i; k < nums.length; k++) {
    nums[k] = 0;
  }
}
